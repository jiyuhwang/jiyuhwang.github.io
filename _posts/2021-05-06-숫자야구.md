		var in1 = document.getElementById("in1");
		var in2 = document.getElementById("in2");
		var in3 = document.getElementById("in3");
		var b = 0;
		var o = 0;
		var s = 0;

		if (arr == "") {
			alert("시작버튼을 누르세요.");

			alert("숫자를 입력하세요.");

				|| isNaN(in3.value) == true) {
			alert("문자가 입력되었습니다. 숫자를 입력해주세요.");

			alert("중복되었습니다. 숫자를 다시 입력해주세요");

			for (var i = 0; i < 3; i++) {
				for (var j = 0; j < 3; j++) {
					if (in4[i] == arr[j] && in4[i] != arr[i]) {
						b++;
					}
				}
				if (in4[i] == arr[i]) {
					s++;
				}
			}
			if (no < 9 && s == 3) {
				alert("승리!!");
			}
			o = 3 - (b + s);

			no++;

					+ b + " O : " + o + ")" + "</div>" + "<hr>";

			wrap.innerHTML += html;

function checkG() {
	var num = document.getElementById("num");
	// 숫자인지 체크
	if(isNaN(num.value * 1)) { //문자가 포함된 경우
		alert("숫자만 입력하세요.");
		num.value = "";
	} else{
		var c = false;
		
		for(var i = 0 ; i < num.value.length - 1 ; i++) {
			for(var j = i + 1 ; j < num.value.length ; j++) {
				if(num.value[i] == num.value[j]) {
					c = true;
					break;
				}
			}
			if(c) {
				break;
			}
		}
		
		if(c) { // c가 true일 때
			alert("중복값이 있습니다.");
			num.value = "";
		} else if(num.value.length == 3) {
			round++;
			
			var o = 0;
			var b = 0;
			var s = 0;
			
			for(var i = 0 ; i < num.value.length ; i++) {
				if(com.indexOf(num.value[i]) == -1) {
					o++;
				} else if(com.indexOf(num.value[i]) == i) {
					s++;
				} else {
					b++;
				}
			}
			
			var t = round + "회 - 입력값 : " + num.value + " [" 
				  + s + "S " + b + "B " + o + "O" + "]";
			
			if(s != 3 && round == 9) {
				var html = "<div class=\"game_over\">" + t + "</div>";
				document.getElementById("res").innerHTML += html;
				num.disabled = true;
			} else if(s != 3) {
				var html = "<div>" + t + "</div>";
				document.getElementById("res").innerHTML += html;
			} else {
				var html = "<div class=\"game_success\">" + t + "</div>";
				document.getElementById("res").innerHTML += html;
				num.disabled = true;
			}
			
			num.value = "";
		}
	} 
}

function set() { // 중복없이 숫자뽑기
	com = "";
	
	for(var i = 0 ; i < 3 ; i++) {
		var a = Math.floor(Math.random() * 9) + 1;
		
		if(com.indexOf(a) == -1) {
			com += a;
		} else {
			i--;
		}
	}
    console.log(com);
}

function newG() {
	set();
	document.getElementById("res").innerHTML = "";
	round = 0;
	document.getElementById("num").disabled = false;
}

function checkG(obj) {
	var num = document.getElementById("num");
	
	obj.disabled = true;
	
	num.value += obj.value;
		
	if(num.value.length == 3) {
		round++;
		
		var o = 0;
		var b = 0;
		var s = 0;
		
		for(var i = 0 ; i < num.value.length ; i++) {
			if(com.indexOf(num.value[i]) == -1) {

			} else {
				b++;

		var t = round + "회 - 입력값 : " + num.value + " [" 
			  + s + "S " + b + "B " + o + "O" + "]";
		
		if(s != 3 && round == 9) {
			var html = "<div class=\"game_over\">" + t + "</div>";
			document.getElementById("res").innerHTML += html;
			num.disabled = true;
			btnDisabled();
		} else if(s != 3) {
			var html = "<div>" + t + "</div>";
			document.getElementById("res").innerHTML += html;
			btnUndisabled();
		} else {
			var html = "<div class=\"game_success\">" + t + "</div>";
			document.getElementById("res").innerHTML += html;
			num.disabled = true;
			btnDisabled();
		}
		
		num.value = "";
	} 
}

function btnUndisabled() {
	for(var i = 1 ; i < 10 ; i++) {
		document.getElementById("btn" + i).disabled = false;
	}
}

function btnDisabled() {
	for(var i = 1 ; i < 10 ; i++) {
		document.getElementById("btn" + i).disabled = true;
	}
}

function set() { // 중복없이 숫자뽑기
	com = "";
	
	for(var i = 0 ; i < 3 ; i++) {
		var a = Math.floor(Math.random() * 9) + 1;
		
		if(com.indexOf(a) == -1) {
			com += a;
		} else {
			i--;
		}
	}
	
	console.log(com);

	btnUndisabled();
