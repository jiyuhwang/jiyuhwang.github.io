```sql
-- 부서별, 업무별 급여 평균을 구하시오.
SELECT DEPTNO, JOB, AVG(SAL)
FROM EMP
GROUP BY DEPTNO, JOB
ORDER BY JOB ASC
;

-- 직원 수가 3명 이상인 업무별 직원 수를 구하시오.
SELECT JOB, CNT
        FROM (SELECT JOB, COUNT(*) AS CNT
        FROM EMP
        GROUP BY JOB)
WHERE CNT >= 3
;

SELECT JOB, COUNT(*) AS CNT
        FROM EMP
        GROUP BY JOB
HAVING COUNT(*) >= 3
;
```

```sql
-- LOCAL_SCOTT
-- 전체 급여 평균 이하로 급여를 받는 사원들의 급여를 30% 인상하여 인상된 급여 기준으로 전체 급여 순위를 구하시오.
-- 내가 푼 방법
SELECT ENAME, SAL, RSAL,
        RANK() OVER(ORDER BY RSAL DESC) AS RNK
FROM
(SELECT ENAME, SAL,
        CASE WHEN  SAL <= (SELECT AVG(SAL) FROM EMP)
             THEN SAL * 1.3
             ELSE SAL
        END AS RSAL
FROM EMP
--GROUP BY ENAME, SAL
) E
;

-- 현님이 푼 방법
SELECT E.ENAME, E.SAL, RANK() OVER(ORDER BY N.UPSAL DESC) AS RNK
FROM EMP E INNER JOIN (SELECT E.ENAME, E.SAL, 
                        CASE WHEN SAL <= S
                        THEN (SAL * 1.3)
                        ELSE SAL
                        END AS UPSAL
                        FROM EMP E INNER JOIN ( SELECT AVG(SAL) AS S FROM EMP ) 
                                    ON  1 = 1) N
                                    ON E.ENAME = N.ENAME
;

SELECT AVG(SAL)
FROM EMP
;

-- 쌤이 푼 방법
SELECT E.ENAME, E.SAL, E.RSAL,
                RANK() OVER(ORDER BY E.RSAL DESC) AS RNK
FROM (SELECT E.ENAME, E.SAL,
        CASE WHEN E.SAL > A.AVG_SAL
            THEN  E.SAL *1.3
            ELSE E.SAL
        END AS RSAL
FROM EMP E INNER JOIN (SELECT ROUND(AVG(SAL)) AS AVG_SAL
                        FROM EMP) A
                    ON 1=1) E
;
```
