	if (a > b) {

		a = b;
		b = temp;
	}
	if (a > c) {
		var temp = a;
		a = c;
		c = temp;
	}
	if (b > c) {
		var temp = b;
		b = c;
		c = temp;

	var arr = [ 5, 7, 3 ];

	for (var i = 0; i < arr.length - 1; i++) {
		for (var j = i + 1; j < arr.length; j++) {
			if (arr[i] < arr[j]) {
				var temp = arr[i];
				arr[i] = arr[j];
				arr[j] = temp;
			}
		}
	}
